import React, { useEffect, useState } from "react";
import SubmitButton from "../SubmitButton/SubmitButton";
import "./AddCourse.css";

export default function AddCourse() {
  // take all of the input into an object and submit it to localStorage
  const [course, SetCourse] = useState({
    courseName: "",
    credits: "",
    professor: "",
    limit: "",
    eligibility: "",
    sem: "",
    courseId: "",
  });



  function handleCoursesSubmit() {
    // update the course in localStorage and setCourse to empty
    let res = JSON.parse(localStorage.getItem("allCourses"));
    res.push(course);
    console.log("added");
    console.log(res);
    localStorage.setItem("allCourses", JSON.stringify(res));
    SetCourse({
      courseName: "",
      credits: "",
      professor: "",
      limit: "",
      eligibility: "",
      sem: "",
      courseId: "",
    });
  }

  function handle(e) {
    let obj = {};
    let a = e.target.name,
      b = e.target.value;

    obj[a] = b;
    SetCourse({
      ...course,
      ...obj,
    });
  }


  return (
    <div className="flex add-course-div">
      <h3>Add Course</h3>
      <input
        name="courseName"
        type="text"
        value = {course.courseName}
        onChange={(e) => {
          SetCourse({ ...course, courseName: e.target.value });
        }}
        placeholder="Enter Course Name"
      />
      <input onChange={handle} val={42424324} name="credits" type="number" placeholder="Enter Credits" />
      <input val={42424324} name="professor" type="text" placeholder="Enter Professor" />
      <input val={42424324} name="limit" type="text" placeholder="Enter Limit" />
      <input val={42424324} name="eligibility" type="text" placeholder="Enter Eligibility" />
      <input val={42424324} name="sem" type="text" placeholder="Enter Sem" />
      <input val={42424324} name="courseId" type="text" placeholder="Enter CourseId" />
      <SubmitButton handleCoursesSubmit={handleCoursesSubmit} />
    </div>
  );
}
